trick 2:
make -f- goal <<< 'goal:;# $(MAKECMDGOALS)'
	-f- 表示从stdin读入数据
	<<<将字符串送到stdin

	等价于
	echo 'goal:;# $(MAKECMDGOALS)' | make -f- goal

	输出：
	# goal
