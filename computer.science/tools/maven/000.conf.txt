    /conf
        settings.xml    # 全局设置


    export M2_HOME=/Users/jiangrui/bin/apache-maven-3.2.1
    export MAVEN_OPTS='-Xms128m -Xmx512m'

~/.m2
    用户目录，保存特定用户的settings.xml和repository
    一般修改这里的settings.xml

目录结构
├── pom.xml
└── src
    ├── main            # 打包到构件中，jar, war
    │   └── java
    │       └── com             # 和pom.xml中groupId对应,com.pekall.mdm
    │           └── juvenxu
    └── test
        └── java        # 不会打包到构件中
            └── com
                └── juvenxu     

settings.xml
    <mirror>设置特定的仓库
